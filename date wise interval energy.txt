select 
       to_char(start_time, 'DD MON YYYY') AS dd,
       to_char(start_time, 'MON DD, YYYY') AS dateWise,
       SUM(CASE WHEN device_device_id = 5012 AND energyConsumption > 0 THEN energyConsumption ELSE 0 END)    AS energy_5012,
       SUM(CASE WHEN device_device_id = 5013 AND energyConsumption > 0 THEN energyConsumption ELSE 0 END)    AS energy_5013,
       SUM(CASE WHEN device_device_id = 5014 AND energyConsumption > 0 THEN energyConsumption ELSE 0 END)    AS energy_5014,
       SUM(CASE WHEN device_device_id = 5015 AND energyConsumption > 0 THEN energyConsumption ELSE 0 END)    AS energy_5015,
       SUM(CASE WHEN device_device_id IN (5012, 5013, 5014, 5015) AND energyConsumption > 0 THEN energyConsumption ELSE 0 END) AS totalEnergy
from (SELECT cd.sl,
             cd.device_device_id,
             start_time,
             end_time,
             case when cd.sl = 1 then cd.energy_max_value - energy_pre_latest_data else energy_consumption end energyConsumption
      FROM (SELECT sl,
                   device_device_id,
                   start_time,
                   end_time,
                   energy_max_value,
                   energy_max_value -
                   nvl((LAG(energy_max_value) OVER (PARTITION BY device_device_id ORDER BY device_device_id, sl)),
                       0) energy_consumption
            FROM (SELECT dtr.sl,
                         dtr.device_device_id,
                         start_time,
                         end_time,
                         LAST_VALUE(max(ehr.ACTIVEENERGYDELIVERED) IGNORE NULLS)
                                    OVER (PARTITION BY device_device_id ORDER BY sl ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) energy_max_value
                  FROM (SELECT d.id device_device_id,
                               d.*,
                               tr.*
                        FROM device_type dt
                                 JOIN device d
                                      ON dt.id = d.device_type_id
                                          AND dt.parent_id = 101
                                          AND d.unit_id = 109
                                          --and d.input_type = 0
                                          AND d.is_active = 1
                                 CROSS JOIN (SELECT LEVEL                                    sl,
                                                    TO_TIMESTAMP(:fromMonth || '00:00:01', 'YYYY-MM-DD HH24:MI:SS') +
                                                    (LEVEL - 1) *
                                                    NUMTODSINTERVAL(1, 'DAY')         AS start_time,
                                                    TO_TIMESTAMP(:fromMonth || '00:00:00', 'YYYY-MM-DD HH24:MI:SS') +
                                                    LEVEL * NUMTODSINTERVAL(1, 'DAY') AS end_time
                                             FROM dual
                                             CONNECT BY LEVEL <=
                                                        (SELECT
                                                             (ADD_MONTHS(TO_DATE(:fromMonth || '00:00:00', 'YYYY-MM-DD HH24:MI:SS'),1) -
                                                              TO_DATE(:fromMonth || '00:00:00', 'YYYY-MM-DD HH24:MI:SS')) *
                                                             1 AS minute_difference
                                                         FROM dual) / 1) tr) dtr
                           LEFT JOIN emeter_health_reading ehr
                                     ON dtr.id = ehr.device_id
                                         AND ehr.date_time BETWEEN dtr.start_time AND dtr.end_time
                                         AND ehr.ACTIVEENERGYDELIVERED > 0
                  group by dtr.sl, dtr.device_device_id, start_time, end_time)) cd
               LEFT JOIN
           (SELECT *
            FROM (SELECT device_id,
                         date_time                                                                                              previous_latest_date_time,
                         ACTIVEENERGYDELIVERED                                                                                     energy_pre_latest_data,
                         ROW_NUMBER() OVER (PARTITION BY device_id ORDER BY device_id, date_time DESC, ACTIVEENERGYDELIVERED DESC) boi_rnk
                  FROM emeter_health_reading
                  WHERE date_time < TO_TIMESTAMP(:fromMonth || '00:00:01', 'YYYY-MM-DD HH24:MI:SS')
                    AND ACTIVEENERGYDELIVERED > 0)
            WHERE boi_rnk = 1) pd
           ON cd.device_device_id = pd.device_id)
group by start_time, end_time
order by start_time
