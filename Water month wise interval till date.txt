----monthwise for generator----------
SELECT 
       to_char(start_time, 'MON YYYY') AS mm,
       to_char(start_time, 'MON YYYY') AS monthWise,
       SUM(CASE
               WHEN device_device_id = 5010 AND waterConsumption > 0 THEN waterConsumption
               ELSE 0 END)                                                                                         AS water_5010,
       SUM(CASE
               WHEN device_device_id = 5016 AND waterConsumption > 0 THEN waterConsumption
               ELSE 0 END)                                                                                         AS water_5016,
       SUM(CASE
               WHEN device_device_id IN (5010, 5016) AND waterConsumption > 0 THEN waterConsumption
               ELSE 0 END)                                                                                         AS totalWater
from (SELECT sl,
                   device_device_id,
                   start_time,
                   end_time,
                   water_pre_latest_data,
                   water_max_value,
                   case
                        when sl = 1 then water_max_value - water_pre_latest_data
                        else water_max_value - nvl((LAG(water_max_value) OVER (PARTITION BY device_device_id ORDER BY device_device_id, sl)),0)
                    end waterConsumption
            FROM (SELECT
                   sl,
                   device_device_id,
                   start_time,
                   end_time,
                   nvl(water_pre_latest_data,0) water_pre_latest_data,
                   LAST_VALUE(case
                        when sl = 1 and water_max_value is null then nvl(water_pre_latest_data,0)
                        else water_max_value
                   end IGNORE NULLS)
                   OVER (PARTITION BY device_device_id ORDER BY sl ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) water_max_value
            FROM (SELECT dtr.sl,
                         dtr.device_device_id,
                         start_time,
                         end_time,
                         LAST_VALUE(max(ehr.WATER_CONSUMPTION) IGNORE NULLS)
                                    OVER (PARTITION BY device_device_id ORDER BY sl ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) water_max_value
                  FROM (SELECT d.id device_device_id,
                               d.*,
                               tr.*
                        FROM device_type dt
                                 JOIN device d
                                      ON dt.id = d.device_type_id
                                          AND dt.parent_id = 104
                                          AND d.unit_id = 109
                                          --and d.input_type = 0
                                          AND d.is_active = 1
                                 CROSS JOIN (SELECT LEVEL AS sl,
                                                       TO_TIMESTAMP(
                                                           TO_CHAR(ADD_MONTHS(TO_DATE(:fromYear, 'DD-MM-YYYY'), LEVEL - 1),
                                                                   'DD-MM-YYYY') || ' 12.00.01.000000000 AM',
                                                           'DD-MM-YYYY HH.MI.SS.FF9 AM') AS start_time,
                                                       TO_TIMESTAMP(
                                                           TO_CHAR(ADD_MONTHS(TO_DATE(:fromYear, 'DD-MM-YYYY'), LEVEL),
                                                                   'DD-MM-YYYY') || ' 12.00.00.000000000 AM',
                                                           'DD-MM-YYYY HH.MI.SS.FF9 AM') AS end_time
                                                FROM dual
                                                CONNECT BY LEVEL <= MONTHS_BETWEEN( ADD_MONTHS(TRUNC(TO_DATE(:toYear, 'DD-MM-YYYY'), 'MM'), 1),
                                                                    TRUNC(TO_DATE(:fromYear, 'DD-MM-YYYY'), 'MM'))
                                                                    ) tr) dtr
                           LEFT JOIN device_reading ehr
                                     ON dtr.id = ehr.device_id
                                         AND ehr.date_time BETWEEN dtr.start_time AND dtr.end_time
                                         AND ehr.date_time <= TO_TIMESTAMP(:toYear || '23:59:59', 'DD-MM-YYYY HH24:MI:SS')
                                         AND ehr.WATER_CONSUMPTION > 0
                  group by dtr.sl, dtr.device_device_id, start_time, end_time) dmv
                  left join (
                            SELECT *
                            FROM (SELECT device_id,
                                         date_time                                                                                                 previous_latest_date_time,
                                         WATER_CONSUMPTION                                                                                     water_pre_latest_data,
                                         ROW_NUMBER() OVER (PARTITION BY device_id ORDER BY device_id, date_time DESC, WATER_CONSUMPTION DESC) boi_rnk
                                  FROM device_reading
                                  WHERE date_time < TO_TIMESTAMP(:fromYear || '00:00:01', 'DD-MM-YYYY HH24:MI:SS')
                                    AND WATER_CONSUMPTION > 0)
                            WHERE boi_rnk = 1
                            ) pd
                            on dmv.device_device_id = pd.device_id))
group by start_time, end_time
order by start_time
